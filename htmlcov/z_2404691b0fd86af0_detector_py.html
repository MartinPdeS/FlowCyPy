77 def capture_signal(self, time: np.ndarray, raw_signal: np.ndarray) -> None: 

78 """ 

79 Captures and processes the raw signal by adding noise, applying baseline shifts, and saturating the signal. 

80 

81 Parameters 

82 ---------- 

83 time : np.ndarray 

84 The time axis of the signal. 

85 raw_signal : np.ndarray 

86 The raw signal before processing. 

87 """ 

88 self.time = time 

89 self.raw_signal = raw_signal 

90 self.signal = raw_signal.copy() 

91 

92 # Add baseline shift 

93 baseline = self.baseline_shift.magnitude * np.sin(0.5 * np.pi * time.magnitude) * ureg.volt 

94 

95 # Add noise 

96 noise = (self.noise_level * np.random.normal(size=len(time))).to('volt') 

97 self.signal += noise + baseline 

98 

99 # Apply saturation 

100 self.signal = np.clip(self.signal.magnitude, 0, self.saturation_level.magnitude) * ureg.volt 

101 

102 # Discretize the signal into bins 

103 bins = np.linspace(np.min(self.signal.magnitude), np.max(self.signal.magnitude), self.n_bins) 

104 digitized = np.digitize(self.signal.magnitude, bins) - 1 

105 self.signal = bins[digitized] * ureg.volt

54 """

56 in a MultiIndex DataFrame with the original units restored. 

57 

58 Parameters 

59 ---------- 

60 peak_detection_func : function 

61 A function that takes in a 2D NumPy array of shape (num_arrays, length_of_array) 

62 and returns a 2D NumPy array (num_arrays, pad_width) containing detected peak indices. 

63 

64 Returns

102 peak_heights_valid = []

129 def _apply_lowpass_filter(self, cutoff_freq: units.Quantity, order: int = 4) -> None: 

130 """ 

131 Applies a Bessel low-pass filter to the signal data in-place. 

132 

133 Parameters 

134 ---------- 

135 cutoff_freq : units.Quantity 

136 The cutoff frequency for the low-pass filter (must have frequency units). 

137 order : int, optional 

138 The order of the Bessel filter, default is 4. 

139 

140 Raises 

141 ------ 

142 ValueError 

143 If cutoff frequency is missing or exceeds Nyquist frequency. 

144 TypeError 

145 If signal data is not stored as a PintArray. 

146 """ 

147 if cutoff_freq is None: 

148 raise ValueError("Cutoff frequency must be specified for low-pass filtering.") 

149 

150 # Get sampling frequency and ensure it's in Hertz 

151 fs = self.parent.cytometer.signal_digitizer.sampling_rate.to("hertz") 

152 nyquist_freq = fs / 2 

153 

154 # Validate cutoff frequency 

155 if cutoff_freq.to("hertz") >= nyquist_freq: 

156 raise ValueError(f"Cutoff frequency ({cutoff_freq}) must be below the Nyquist frequency ({nyquist_freq}).") 

157 

158 # Ensure the signal column is a PintArray 

159 if not isinstance(self.analog["Signal"].array, pint_pandas.PintArray): 

160 raise TypeError("Expected 'Signal' to be a PintArray, but got a different type.") 

161 

162 # Iterate through each detector-segment pair and apply filtering 

163 for (detector, segment_id), segment_data in self.analog.groupby(level=['Detector', 'SegmentID']): 

164 filtered_values = bessel_lowpass_filter( 

165 signal=segment_data["Signal"].pint.quantity.magnitude, 

166 cutoff=cutoff_freq, 

167 sampling_rate=fs, 

168 order=order 

169 ) 

170 

171 self.analog.loc[(detector, segment_id), "Signal"] = filtered_values
